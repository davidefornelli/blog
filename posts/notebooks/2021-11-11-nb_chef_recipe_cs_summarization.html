Import ingredients

def ingredients_import(ingredients):
    for ingredient in ingredients:
        mod, package = chef.process_gist_ingredient(
            gist_id=ingredient[0],
            gist_file=ingredient[1],
            gist_user=gist_user
        )
        globals()[package] = mod


ingredients_import(ingredients=ingredients_to_import)
